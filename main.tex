\section{Scope and Positioning}
\label{sec:scope}

We fix one quantile level $p\in(0,1)$ and work conditionally on deterministic DESN features
$\{\vect{x}_t\}_{t=1}^T$, $\vect{x}_t\in\R^{k}$.
The default shrinkage framework in this document is the
\emph{Nishimura--Suchard-style regularized horseshoe construction} (denoted here as
\texttt{rhs\_ns}) \cite{nishimura-suchard,piironen-vehtari}.

Our objective is implementation-ready derivation of:
\begin{enumerate}[leftmargin=1.7em]
  \item full posterior structure,
  \item MCMC blocks and full conditionals,
  \item VB-CAVI factorization and updates,
  \item ELBO decomposition with exact-vs-approximate separation.
\end{enumerate}

\section{Model Setup (Q--DESN with exAL Augmentation)}
\label{sec:model}

\subsection{Likelihood and Latent Augmentation}

For each $t=1,\dots,T$,
\begin{align}
  y_t\mid\vect{\beta},\sigma,\gamma,v_t,s_t
  &\sim \N\!\Big(m_t,\,\tau_t^2\Big),
  \label{eq:y-cond}\\
  m_t
  &= \vect{x}_t^\top\vect{\beta}
     + \sigma C(\gamma)|\gamma|\,s_t
     + A(\gamma)\,v_t,
  \label{eq:mean-def}\\
  \tau_t^2
  &= \sigma B(\gamma) v_t,
  \label{eq:var-def}
\end{align}
with latent variables
\begin{align}
  v_t\mid\sigma &\sim \Exp(\text{mean}=\sigma),
  &
  s_t &\sim \TN(0,1).
\end{align}
Here $A(\gamma),B(\gamma),C(\gamma)$ are the exAL functions at fixed quantile level $p$ \cite{yan-exal}.

\subsection{Parameter Blocks}

We use $\vect{\beta}=(\beta_0,\beta_1,\dots,\beta_r)^\top$ with $r=k-1$.
The intercept has an independent Gaussian prior:
\begin{equation}
  \beta_0 \sim \N(b_0,v_0), \qquad v_0>0.
\end{equation}
For $j=1,\dots,r$, coefficients are under the \texttt{rhs\_ns} shrinkage block (Section~\ref{sec:rhs-ns}).
We also place
\begin{align}
  \sigma &\sim \IG(a_\sigma,b_\sigma),
  &
  \gamma &\sim \pi_\gamma(\gamma)\,\mathbf{1}(L<\gamma<U),
\end{align}
with fixed support interval $(L,U)$ from the quantile-identifying exAL construction.

\section{Default Prior: \texttt{rhs\_ns} (and Legacy Context)}
\label{sec:rhs-ns}

\subsection{Coefficient Conditional Law and Effective Variance}

Define local scales $\lambda_j>0$, global scale $\tau>0$, and slab scale $\zeta>0$.
The regularized conditional Gaussian law is
\begin{equation}
  \beta_j\mid\lambda_j,\tau,\zeta
  \sim \N\!\left(0,\,V_j\right),
  \qquad
  V_j = \left(\zeta^{-2}+\tau^{-2}\lambda_j^{-2}\right)^{-1}
      = \frac{\zeta^2\tau^2\lambda_j^2}{\zeta^2+\tau^2\lambda_j^2}.
  \label{eq:rhs-cond-beta}
\end{equation}
This is the same conditional variance object used in both PV and NS formulations.

\subsection{\texttt{rhs\_ns} Joint Construction}

The \texttt{rhs\_ns} default is the joint representation
\begin{equation}
  p_{\mathrm{ns}}(\beta_j,\lambda_j\mid\tau,\zeta)
  \propto
  \exp\!\left(-\frac{\beta_j^2}{2\tau^2\lambda_j^2}\right)
  \exp\!\left(-\frac{\beta_j^2}{2\zeta^2}\right)
  \lambda_j^{-1}(1+\lambda_j^2)^{-1},
  \label{eq:ns-joint-product}
\end{equation}
which implies \eqref{eq:rhs-cond-beta} for $\beta_j\mid\lambda_j,\tau,\zeta$ and preserves the ordinary horseshoe
full-conditional global-local block given $\beta$ \cite{nishimura-suchard}.

Equivalent conditional-prior form:
\begin{equation}
  p_{\mathrm{ns}}(\lambda_j\mid\tau,\zeta)
  \propto
  (1+\lambda_j^2)^{-1}
  \left(1+\frac{\tau^2\lambda_j^2}{\zeta^2}\right)^{-1/2}.
  \label{eq:ns-induced-local-prior}
\end{equation}

\subsection{Pseudo-data Interpretation}

Introduce pseudo-observations $z_j=0$ with
\begin{equation}
  z_j\mid\beta_j,\zeta^2 \sim \N(\beta_j,\zeta^2),
\end{equation}
which contributes $\exp\{-\beta_j^2/(2\zeta^2)\}$ and recovers
\eqref{eq:ns-joint-product}. This makes the algorithmic role of $\zeta^{-2}$ explicit as an added ridge precision on $\beta$.

\subsection{Legacy Prior (Context Only, Not Default)}

The direct PV regularized prior keeps the original local-scale prior and therefore induces an extra
$\big(1+\tau^2\lambda_j^2/\zeta^2\big)^{1/2}$ factor in
$p(\tau,\lambda\mid\beta,\cdot)$.
Hence:
\begin{itemize}[leftmargin=1.7em]
  \item \textbf{conditional equivalence}: PV and NS share \eqref{eq:rhs-cond-beta};
  \item \textbf{joint-prior non-equivalence}: PV and NS are different joint priors;
  \item \textbf{computational consequence}: NS preserves the ordinary full-conditional global-local block given $\beta$, PV does not.
\end{itemize}
This distinction is central for implementation design \cite{piironen-vehtari,nishimura-suchard,bhattacharya-khare-pal}.

\section{Full Posterior Structure}
\label{sec:posterior}

Let
\begin{equation}
  \Theta = (\vect{\beta},\vect{\lambda},\tau,\zeta,\sigma,\gamma),
  \qquad
  \vect{z}_{1:T} = \{(v_t,s_t)\}_{t=1}^T,
  \qquad
  \vect{z}^{\mathrm{ps}} = (z_1,\dots,z_r)=(0,\dots,0).
\end{equation}

Under \texttt{rhs\_ns} (pseudo-data view), up to normalizing constant,
\begin{equation}
\begin{aligned}
  p(\Theta,\vect{z}_{1:T}\mid\vect{y},\vect{z}^{\mathrm{ps}})
  &\propto
  \prod_{t=1}^T p\big(y_t\mid\vect{\beta},\sigma,\gamma,v_t,s_t\big)
  \prod_{t=1}^T p(v_t\mid\sigma)p(s_t)
  \prod_{j=1}^r p(z_j=0\mid\beta_j,\zeta^2)
  \\
  &\quad\times p(\beta_0)
  \prod_{j=1}^r p(\beta_j\mid\lambda_j,\tau)
  \;p(\vect{\lambda},\tau)\;p(\zeta^2)\;p(\sigma,\gamma).
\end{aligned}
\label{eq:posterior-factorization}
\end{equation}

For exact global-local conjugate updates we use the auxiliary inverse-gamma representation
\cite{makalic-schmidt}:
\begin{align}
  \lambda_j^2\mid\nu_j &\sim \IG\!\left(\frac12,\frac{1}{\nu_j}\right), &
  \nu_j &\sim \IG\!\left(\frac12,1\right),\\
  \tau^2\mid\xi &\sim \IG\!\left(\frac12,\frac{1}{\xi}\right), &
  \xi &\sim \IG\!\left(\frac12,\frac{1}{\tau_0^2}\right),\\
  \zeta^2 &\sim \IG(a_\zeta,b_\zeta) \quad \text{(optional; can be fixed).}
\end{align}

\section{MCMC Blocks and Full Conditionals}
\label{sec:mcmc}

\subsection{Block Overview}

A practical blocked sampler updates:
\begin{enumerate}[leftmargin=1.7em]
  \item $\vect{\beta}$,
  \item $(\lambda_1^2,\nu_1),\dots,(\lambda_r^2,\nu_r),\tau^2,\xi$,
  \item $\zeta^2$ (if random),
  \item $(\sigma,\gamma)$,
  \item $v_{1:T}$,
  \item $s_{1:T}$.
\end{enumerate}

\subsection{Exact Closed-form Blocks under \texttt{rhs\_ns} + IG Augmentation}

Define
\begin{align}
  y_t^\star &= y_t - \sigma C(\gamma)|\gamma|s_t - A(\gamma)v_t,
  &
  w_t &= \frac{1}{\sigma B(\gamma)v_t},
\end{align}
and let $\mat{W}=\diag(w_1,\dots,w_T)$,
$\vect{y}^\star=(y_1^\star,\dots,y_T^\star)^\top$.

\paragraph{(i) Coefficients $\vect{\beta}$.}
\begin{equation}
  \vect{\beta}\mid\text{rest}
  \sim
  \N\!\big(\vect{\mu}_\beta,\mat{\Sigma}_\beta\big),
\end{equation}
with
\begin{align}
  \mat{\Sigma}_\beta^{-1}
  &= \mat{X}^\top\mat{W}\mat{X} + \mat{D}^{-1},
  \\
  \vect{\mu}_\beta
  &= \mat{\Sigma}_\beta\Big(\mat{X}^\top\mat{W}\vect{y}^\star + \mat{D}^{-1}\vect{b}\Big),
\end{align}
$\vect{b}=(b_0,0,\dots,0)^\top$, and
\begin{equation}
  \mat{D}^{-1}
  =
  \diag\!\left(
  v_0^{-1},
  \tau^{-2}\lambda_1^{-2}+\zeta^{-2},\dots,
  \tau^{-2}\lambda_r^{-2}+\zeta^{-2}
  \right).
\end{equation}

\paragraph{(ii) Local scales.}
For $j=1,\dots,r$:
\begin{align}
  \lambda_j^2\mid\text{rest}
  &\sim
  \IG\!\left(
    1,
    \frac{1}{\nu_j} + \frac{\beta_j^2}{2\tau^2}
  \right),
  \label{eq:fc-lambda2}
  \\
  \nu_j\mid\text{rest}
  &\sim
  \IG\!\left(
    1,
    1 + \frac{1}{\lambda_j^2}
  \right).
  \label{eq:fc-nu}
\end{align}

\paragraph{(iii) Global scale.}
\begin{align}
  \tau^2\mid\text{rest}
  &\sim
  \IG\!\left(
    \frac{r+1}{2},
    \frac{1}{\xi} + \frac12\sum_{j=1}^r \frac{\beta_j^2}{\lambda_j^2}
  \right),
  \label{eq:fc-tau2}
  \\
  \xi\mid\text{rest}
  &\sim
  \IG\!\left(
    1,
    \frac{1}{\tau_0^2} + \frac{1}{\tau^2}
  \right).
  \label{eq:fc-xi}
\end{align}

\paragraph{(iv) Slab scale (if random).}
Because pseudo-data enter as
$\prod_{j=1}^r \N(0\mid\beta_j,\zeta^2)$,
\begin{equation}
  \zeta^2\mid\text{rest}
  \sim
  \IG\!\left(
    a_\zeta + \frac{r}{2},
    b_\zeta + \frac12\sum_{j=1}^r \beta_j^2
  \right).
  \label{eq:fc-zeta2}
\end{equation}

\paragraph{(v) exAL latent $v_t$.}
Let
\begin{align}
  \delta_t
  &= y_t - \vect{x}_t^\top\vect{\beta} - \sigma C(\gamma)|\gamma|s_t,
  \\
  \psi_t
  &= \frac{A(\gamma)^2}{\sigma B(\gamma)} + \frac{2}{\sigma},
  \qquad
  \chi_t = \frac{\delta_t^2}{\sigma B(\gamma)}.
\end{align}
Then
\begin{equation}
  v_t\mid\text{rest}
  \sim
  \GIG\!\left(\frac12,\chi_t,\psi_t\right).
  \label{eq:fc-v}
\end{equation}

\paragraph{(vi) exAL latent $s_t$.}
Define
\begin{align}
  \mu_{0t} &= y_t - \vect{x}_t^\top\vect{\beta} - A(\gamma)v_t,
  &
  C_\gamma &= C(\gamma)|\gamma|,
\end{align}
\begin{align}
  a_t &= 1 + \frac{\sigma C_\gamma^2}{B(\gamma)v_t},
  &
  m_t^{(s)} &= \frac{\mu_{0t}C_\gamma}{B(\gamma)v_t+\sigma C_\gamma^2}.
\end{align}
Then
\begin{equation}
  s_t\mid\text{rest}
  \sim
  \N\!\big(m_t^{(s)},a_t^{-1}\big)\;\text{truncated to }(0,\infty).
  \label{eq:fc-s}
\end{equation}

\subsection{Approximate / Numerical Blocks}

\paragraph{$(\sigma,\gamma)$ block.}
The full conditional is smooth but non-conjugate:
\begin{equation}
  p(\sigma,\gamma\mid\text{rest})
  \propto
  \exp\{\ell_{\sigma,\gamma}(\sigma,\gamma)\}\,
  \mathbf{1}(\sigma>0,\,L<\gamma<U),
\end{equation}
with
\begin{equation}
\begin{aligned}
  \ell_{\sigma,\gamma}
  &= -\frac12\sum_{t=1}^T\log\big(\sigma B(\gamma)v_t\big)
     -\frac12\sum_{t=1}^T
      \frac{\big(y_t-m_t\big)^2}{\sigma B(\gamma)v_t}
     -\sum_{t=1}^T\log\sigma
     -\sum_{t=1}^T\frac{v_t}{\sigma}
  \\
  &\quad -(a_\sigma+1)\log\sigma - \frac{b_\sigma}{\sigma}
     +\log\pi_\gamma(\gamma).
\end{aligned}
\label{eq:sigmagamma-kernel}
\end{equation}
Use MH, slice, or HMC on transformed coordinates.

\paragraph{Direct-half-Cauchy implementation without IG augmentation.}
If one does not augment half-Cauchy scales, updates for $(\vect{\lambda},\tau)$ become non-standard and require numerical moves.
Under \texttt{rhs\_ns}, this is a computational choice, not a change in target posterior.

\section{VB-CAVI under \texttt{rhs\_ns}}
\label{sec:vb}

\subsection{Mean-field Factorization}

We use
\begin{equation}
\begin{aligned}
  q(\Theta,\vect{z}_{1:T})
  &= q_\beta(\vect{\beta})
     \left[\prod_{j=1}^r q_{\lambda_j^2}(\lambda_j^2)q_{\nu_j}(\nu_j)\right]
     q_{\tau^2}(\tau^2)q_\xi(\xi)
     q_{\zeta^2}(\zeta^2)
  \\
  &\quad\times q_{\sigma,\gamma}(\sigma,\gamma)
     \prod_{t=1}^T q_{v_t}(v_t)q_{s_t}(s_t),
\end{aligned}
\label{eq:vb-factorization}
\end{equation}
where $q_{\zeta^2}$ is omitted when $\zeta$ is fixed.

\subsection{Closed-form CAVI Updates}

\paragraph{(i) $q_\beta$.}
\begin{equation}
  q_\beta(\vect{\beta}) = \N(\vect{\mu}_\beta,\mat{\Sigma}_\beta),
\end{equation}
with
\begin{align}
  \mat{\Sigma}_\beta
  &= \left(\mat{X}^\top\bar{\mat{W}}\mat{X} + \bar{\mat{D}}^{-1}\right)^{-1},
  \label{eq:vb-beta-sigma}
  \\
  \vect{\mu}_\beta
  &= \mat{\Sigma}_\beta\left(\mat{X}^\top\bar{\vect{m}} + \bar{\mat{D}}^{-1}\vect{b}\right),
  \label{eq:vb-beta-mu}
\end{align}
where
\begin{align}
  \bar{w}_t &= \E_q\!\left[\frac{1}{\sigma B(\gamma)v_t}\right],
  &
  \bar{\mat{W}} &= \diag(\bar{w}_1,\dots,\bar{w}_T),
  \\
  \bar{m}_t &= \E_q\!\left[
    \frac{y_t-\sigma C(\gamma)|\gamma|s_t-A(\gamma)v_t}{\sigma B(\gamma)v_t}
  \right],
  &
  \bar{\vect{m}} &= (\bar{m}_1,\dots,\bar{m}_T)^\top,
\end{align}
and
\begin{equation}
  \bar{\mat{D}}^{-1}
  = \diag\!\left(
    v_0^{-1},
    \E_q[\tau^{-2}]\E_q[\lambda_1^{-2}] + \E_q[\zeta^{-2}],
    \dots,
    \E_q[\tau^{-2}]\E_q[\lambda_r^{-2}] + \E_q[\zeta^{-2}]
  \right).
\end{equation}

\paragraph{(ii) Scale-mixture factors.}
For $j=1,\dots,r$,
\begin{align}
  q_{\lambda_j^2}
  &= \IG\!\left(
      1,
      \E_q[\nu_j^{-1}] + \frac12\E_q[\beta_j^2]\E_q[\tau^{-2}]
    \right),
  \label{eq:vb-lambda2}
  \\
  q_{\nu_j}
  &= \IG\!\left(
      1,
      1 + \E_q[\lambda_j^{-2}]
    \right),
  \label{eq:vb-nu}
\end{align}
\begin{align}
  q_{\tau^2}
  &= \IG\!\left(
      \frac{r+1}{2},
      \E_q[\xi^{-1}] + \frac12\sum_{j=1}^r \E_q[\beta_j^2]\E_q[\lambda_j^{-2}]
    \right),
  \label{eq:vb-tau2}
  \\
  q_\xi
  &= \IG\!\left(
      1,
      \frac{1}{\tau_0^2} + \E_q[\tau^{-2}]
    \right).
  \label{eq:vb-xi}
\end{align}
If $\zeta^2$ is random:
\begin{equation}
  q_{\zeta^2}
  = \IG\!\left(
      a_\zeta + \frac{r}{2},
      b_\zeta + \frac12\sum_{j=1}^r \E_q[\beta_j^2]
    \right).
  \label{eq:vb-zeta2}
\end{equation}

\paragraph{(iii) exAL latent factors.}
\begin{align}
  q_{v_t}(v_t)
  &= \GIG\!\left(\frac12,\bar\chi_t,\bar\psi_t\right),
  \label{eq:vb-vt}
  \\
  q_{s_t}(s_t)
  &= \N\!\big(m_t^{(0)},(\bar a_t)^{-1}\big)\;\text{truncated to }(0,\infty),
  \label{eq:vb-st}
\end{align}
with
\begin{align}
  \bar\psi_t &= \E_q\!\left[\frac{A(\gamma)^2}{\sigma B(\gamma)}+\frac{2}{\sigma}\right],
  &
  \bar\chi_t &= \E_q\!\left[\frac{\delta_t^2}{\sigma B(\gamma)}\right],
  \\
  \delta_t &= y_t-\vect{x}_t^\top\vect{\beta}-\sigma C(\gamma)|\gamma|s_t,
  \\
  \bar a_t &= 1+\E_q\!\left[\frac{\sigma C(\gamma)^2\gamma^2}{B(\gamma)v_t}\right],
  &
  \bar b_t &= \E_q\!\left[\frac{(y_t-\vect{x}_t^\top\vect{\beta}-A(\gamma)v_t)C(\gamma)|\gamma|}{B(\gamma)v_t}\right],
\end{align}
and $m_t^{(0)}=\bar b_t/\bar a_t$.

\subsection{Approximate CAVI Block: $q_{\sigma,\gamma}$}

The $(\sigma,\gamma)$ factor is non-conjugate.
Use transformed coordinates
$\rho=\log\sigma$ and $\xi_\gamma=\log\frac{\gamma-L}{U-\gamma}$,
and approximate
\begin{equation}
  q_{\sigma,\gamma}(\rho,\xi_\gamma)
  \approx
  \N_2\!\big(\hat{\vect\eta}_{\sigma,\gamma},\mat{H}_{\sigma,\gamma}^{-1}\big)
\end{equation}
by Laplace at the mode of
$\E_{q_{-(\sigma,\gamma)}}[\log p(\vect{y},\vect{z},\Theta)]$
plus Jacobian term.
Expectations coupling to this block are computed via second-order delta approximations.

\section{ELBO Decomposition and Exact-vs-Approximate Terms}
\label{sec:elbo}

With \eqref{eq:vb-factorization},
\begin{equation}
\begin{aligned}
  \mathcal{L}(q)
  &= \E_q\![\log p(\vect{y}\mid\cdot)]
   + \E_q\![\log p(\vect{v}\mid\sigma)]
   + \E_q\![\log p(\vect{s})]
   + \E_q\![\log p(\vect{\beta}\mid\vect{\lambda},\tau,\zeta)]
  \\
  &\quad
   + \E_q\![\log p(\vect{\lambda}^2\mid\vect\nu)]
   + \E_q\![\log p(\vect\nu)]
   + \E_q\![\log p(\tau^2\mid\xi)]
   + \E_q\![\log p(\xi)]
   + \E_q\![\log p(\zeta^2)]
   + \E_q\![\log p(\sigma,\gamma)]
  \\
  &\quad
   - \E_q\![\log q_\beta]
   - \sum_{j=1}^r\E_q\![\log q_{\lambda_j^2}]
   - \sum_{j=1}^r\E_q\![\log q_{\nu_j}]
   - \E_q\![\log q_{\tau^2}]
   - \E_q\![\log q_\xi]
   - \E_q\![\log q_{\zeta^2}]
  \\
  &\quad
   - \E_q\![\log q_{\sigma,\gamma}]
   - \sum_{t=1}^T\E_q\![\log q_{v_t}]
   - \sum_{t=1}^T\E_q\![\log q_{s_t}].
\end{aligned}
\label{eq:elbo-full}
\end{equation}

\paragraph{Exact ELBO pieces.}
All terms involving Gaussian, inverse-gamma, GIG, and truncated-normal factors are closed form (or numerically exact through standard special-function calls for GIG and truncated-normal entropy).

\paragraph{Approximate ELBO pieces.}
Terms depending on $(\sigma,\gamma)$ and mixed nonlinear quantities (e.g.,
$\E_q[(y_t-m_t)^2/(\sigma B(\gamma)v_t)]$) are evaluated by Laplace--Delta expansions around the current mode of $q_{\sigma,\gamma}$.

\section{Algorithmic Summary for Implementation}
\label{sec:algorithms}

\subsection{Gibbs/Laplace Hybrid Sampler}
\begin{enumerate}[leftmargin=1.7em]
  \item Update $v_{1:T}$ by \eqref{eq:fc-v} and $s_{1:T}$ by \eqref{eq:fc-s}.
  \item Update $\vect{\beta}$ by the Gaussian block in Section~\ref{sec:mcmc}.
  \item Update $(\lambda_j^2,\nu_j)_{j=1}^r$, $\tau^2$, and $\xi$ via
  \eqref{eq:fc-lambda2}--\eqref{eq:fc-xi}.
  \item If random slab is used, update $\zeta^2$ by \eqref{eq:fc-zeta2}; otherwise keep fixed.
  \item Update $(\sigma,\gamma)$ numerically from \eqref{eq:sigmagamma-kernel} on transformed coordinates.
\end{enumerate}

\subsection{VB-CAVI + Laplace-Delta}
\begin{enumerate}[leftmargin=1.7em]
  \item Update $q_\beta$ by \eqref{eq:vb-beta-sigma}--\eqref{eq:vb-beta-mu}.
  \item Update $q_{v_t}$ and $q_{s_t}$ for all $t$ by \eqref{eq:vb-vt}--\eqref{eq:vb-st}.
  \item Update $q_{\lambda_j^2},q_{\nu_j},q_{\tau^2},q_\xi$ (and $q_{\zeta^2}$ if random) by
  \eqref{eq:vb-lambda2}--\eqref{eq:vb-zeta2}.
  \item Update $q_{\sigma,\gamma}$ via Laplace on transformed coordinates.
  \item Recompute the ELBO \eqref{eq:elbo-full} and stop when relative increment is below tolerance.
\end{enumerate}

\section{Key Technical Distinctions (Explicit)}
\label{sec:distinctions}

\begin{itemize}[leftmargin=1.7em]
  \item \textbf{Exact closed-form updates:} $\vect{\beta}$, $(\lambda_j^2,\nu_j)$, $(\tau^2,\xi)$,
  $\zeta^2$ (if random), $v_t$, and $s_t$ under the augmented hierarchy.
  \item \textbf{Approximate / numerical updates:} $(\sigma,\gamma)$ in both MCMC and VB-CAVI.
  \item \textbf{Conditional equivalence:} PV and NS share
  $\beta_j\mid\lambda_j,\tau,\zeta$.
  \item \textbf{Joint-prior non-equivalence:} NS is not the PV joint prior; NS is used here as default because it preserves the full-conditional global-local block given $\beta$.
\end{itemize}
